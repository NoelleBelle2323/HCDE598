				function drawCloud(x, y, opacity) {
					noStroke();
					fill(255, 255, 255, opacity);
					ellipse(x, y, 100, 100);
					ellipse(x + 20, y + 20, 100, 50);
					ellipse(x - 10, y + 5, 140, 60);
				}

				function drawMist(x, y, opacity) {
					fill(255, 255, 255, opacity);
					ellipse(x, y, 300, 40);
				}

				function drawBigmist(x, y, opacity) {
					fill(255, 255, 255, opacity);
					ellipse(x, y, 200, 100);
				}
