				//function to draw the left arrow key
				function keyButton1(x, y) {
				  push();
				  noStroke();
				  fill(55, 78, 94); //dark blue
				  rect(x, y, 50, 50);
				  fill(100, 141, 171); //light blueish
				  rect(x + 5, y + 5, 35, 35);
				  fill(171, 53, 50); //dark red
				  triangle(x + 6, y + 20, x + 15, y + 6, x + 15, y + 35);
				  stroke(171, 53, 50);
				  strokeWeight(5);
				  line(x + 15, y + 20, x + 30, y + 20);
				  pop();
				}
