				//Emoji

				/** I took the animation from assignment 2 and created conditions for animation
					* If 1 is pressed, draw the emoji
					* If 2 is pressed, the emojis will move to the center
				  * If 3 is pressed, the emojis will rotate
					*/	

				//Setting up variable
				var x1 = 50;
				var y1 = 50;
				var x2 = 550;
				var y2 = 50;
				var x3 = 50;
				var y3 = 450;
				var x4 = 550;
				var y4 = 450;
				var shapewidth = 70;
				var shapeheight = 70;
				var text1 = "Press 1 to draw emojis";
				var text2 = "Press 2 to move emojis to center";
				var text3 = "Press 3 to rotate emojis";
				var text4 = "You did it!";
				var onPage1 = true;
				var onPage2 = false;

				var angle = 0;


				function setup() {
				  createCanvas(600, 500);
				  angleMode(DEGREES);
				  ellipseMode(CENTER);
				}

				function draw() {

				  if(onPage1){
				  	background(220);
						
				    TextStyle();
				  
						text(text1, 300,450);
				  
						//When 1 is pressed
				  	if (key == 1){
				    
				      onPage1 = false;
				      onPage2 = true;
				      CoverOldText();
				      DrawEmojis(); //Draw emojis
				      TextStyle(); //Define text style
				      text(text2, 300,450);//Display new texts
				  	} 
				    
				  }
				    
				  	//When 2 is pressed
				  	if (key == 2){
				    background(220);
				    MoveToCenter();
				    TextStyle();
				    text(text3, 300,450);//Display new texts
					
				    }
				  
				  //When 3 is pressed
				  	if(key == 3){
				    background(220);
				    RotateEmojis();
				    TextStyle();
				    text(text4,300,450);//Display new texts
				    
				    } 
				     

				  
				//Specify text styles
				function TextStyle(){
					fill(0);
				  textFont('Helvetica');
				  textSize(20);
					textAlign(CENTER);
				}
				  
				//Rectangle to cover old texts
				function CoverOldText(){
				    noStroke();
				  	fill(220);
				  	rect(180,430,250,40);
				}

				//////////////////////////////////
				/////Move Emojis to Center////////
				///////////////////////////////// 
				function MoveToCenter() {
				  
				  DrawEmojis();
				  
				  //Movement - top left emoji
				  if (x1<150){
				  	x1 = x1+5;  
				  }
				  
				  	else{
				    	y1 = y1+ 5;
				  }
				  
				  if(y1>250){
				    y1=250;
				  }
				  
					//Movement - top right emoji
				  
				  if(x2>350){
				    x2 = x2-5;
				  }
				  
				  else{
				  	y2 = y2+5;
				  }
				  
				  if(y2>250){
				    y2 = 250;
				  }

					//Movement - bottom left emoji
				  if(x3<250){
				    x3 = x3+5;
				  }
				  
				  else{
				    y3 = y3-5;
				  }
				  
				  if(y3<250){
				    y3 = 250;
				  }
				  
					//Movement - bottom right emoji
				  if(x4>450){
				    x4=x4-5;
				  }
				  else
				    y4= y4-5
				  
				  if(y4<250){
				    y4=250;
				  }

					}
